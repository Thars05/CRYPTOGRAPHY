<script>

    function gcdIterative(a, b) {
        a = Math.abs(a);
        b = Math.abs(b);
        while (b !== 0) {
            let remainder = a % b;
            a = b;
            b = remainder;
        }
        return a;
    }


    function calculateGCD() {

        const val1 = parseInt(document.getElementById('num1').value);
        const val2 = parseInt(document.getElementById('num2').value);
        const resultDiv = document.getElementById('result');

        if (isNaN(val1) || isNaN(val2)) {
            resultDiv.innerText = "Please enter two valid numbers.";
            resultDiv.style.color = "red";
            return;
        }

        const gcd = gcdIterative(val1, val2);


        resultDiv.style.color = "#333";
        resultDiv.innerText = `The GCD of ${val1} and ${val2} is: ${gcd}`;
    }
</script>
