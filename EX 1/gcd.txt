        <button onclick="calculateSubtractionGCD()">Calculate GCD</button>

    <script>
        function calculateSubtractionGCD() {
            let a = Math.abs(parseInt(document.getElementById('num1').value));
            let b = Math.abs(parseInt(document.getElementById('num2').value));

            if (isNaN(a) || isNaN(b)) {
                document.getElementById('result').innerText = "Please enter valid numbers.";
                return;
            }
            if (a === 0) { a = b; }
            else if (b !== 0) {
                // GCD Algorithm (Subtraction Method)
                while (a !== b) {
                    if (a > b) a = a - b;
                    else b = b - a;
                }
            }

            document.getElementById('result').innerText = "GCD is: " + a;
        }
    </script>
