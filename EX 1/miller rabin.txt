<script>
    /**
     * Modular Exponentiation: (base^exp) % mod
     * Uses BigInt for precision with large numbers.
     */
    function power(base, exp, mod) {
        let res = 1n;
        base = base % mod;
        while (exp > 0n) {
            if (exp % 2n === 1n) res = (res * base) % mod;
            base = (base * base) % mod;
            exp = exp / 2n;
        }
        return res;
    }

    /**
     * Single round of the Miller-Rabin test.
     */
    function millerTest(d, n) {
        // Pick a random witness 'a' in range [2, n-2]
        const randomRange = n - 4n;
        // Simple random BigInt generation
        const a = 2n + BigInt(Math.floor(Math.random() * 1000000)) % randomRange;

        let x = power(a, d, n);
        if (x === 1n || x === n - 1n) return true;

        while (d !== n - 1n) {
            x = (x * x) % n;
            d *= 2n;
            if (x === 1n) return false;
            if (x === n - 1n) return true;
        }
        return false;
    }

    /**
     * Checks if n is probably prime using k iterations.
     */
    function isPrime(n, k) {
        if (n <= 1n || n === 4n) return false;
        if (n <= 3n) return true;
        if (n % 2n === 0n) return false;

        let d = n - 1n;
        while (d % 2n === 0n) d /= 2n;

        for (let i = 0; i < k; i++) {
            if (!millerTest(d, n)) return false;
        }
        return true;
    }

    function runPrimalityTest() {
        const nStr = document.getElementById('nInput').value.trim();
        const kValue = parseInt(document.getElementById('kInput').value);
        const display = document.getElementById('result');

        try {
            const n = BigInt(nStr);
            const startTime = performance.now();
            const result = isPrime(n, kValue);
            const endTime = performance.now();

            display.className = result ? "is-prime" : "is-composite";
            display.innerText = result ?
                `PROBABLY PRIME (checked in ${(endTime - startTime).toFixed(2)}ms)` :
                "DEFINITELY COMPOSITE";
        } catch (e) {
            display.className = "is-composite";
            display.innerText = "Error: Please enter a valid integer.";
        }
    }
